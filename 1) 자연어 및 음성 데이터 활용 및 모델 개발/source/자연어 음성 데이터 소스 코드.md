## 페이지 18 - 텍스트 전처리 파이프라인

```python
import re 
from konlpy.tag import Okt

okt = Okt()

RE_PUNCT = re.compile(r"[\^\w\s]", flags=re.UNICODE) 
RE_NUM = re.compile(r"\d+")

def preprocess_korean_text(text, keep_num=False, num_token=" <NUM>"):
    text = text.lower() # (a) 소문자화 
    text = RE_PUNCT.sub(" ", text) # (b) 특수문자/구두점 처리 
    if keep_num: # (c) 숫자 처리 
        text = RE_NUM.sub(num_token, text) 
    else: 
        text = RE_NUM.sub(" ", text)

    tokens = okt.morphs(text) # (d) 형태소 분석 (토큰화) 
    stopwords = {"은", "는", "이", "가", "을", "를", "에", "의", "으로", "도", "에서"} # (e) 불용어 제거 
    tokens = [t for t in tokens if t not in stopwords and t.strip()]

# 예시 텍스트 
sample_text = "오늘은 날씨가 좋아서 공원에서 산책을 했습니다. 사람들 200 명이 벚꽃을 구경했어요!"

# 전처리 적용 
processed = preprocess_korean_text(sample_text, keep_num=True) 
print(processed)
```

```python
print("Kkma 품사 태깅:", kkma.pos(text))

times = {} 
for name, analyzer in [("OKT", okt), ("Komoran", komoran), ("Kkma", kkma)]:

        analyzer.morphs(text)

for name, elapsed in times.items():  
    print(f"{name}: {elapsed:.4f}초")
```

```python
    "안녕하세요 자연어처리를 배우고 있습니다", 
    "자연어처리는 인공지능의 중요한 분야입니다", 
    "토큰화는 자연어처리의 기본 과정입니다", 
    "BPE는 자주 등장하는 문자열을 하나의 토큰으로 만듭니다" 
]

from tokenizers import Tokenizer 
from tokenizers.models import BPE 
from tokenizers.trainers import BpeTrainer 
from tokenizers.pre_tokenizers import Whitespace

# BPE 토크나이저 생성 
tokenizer = Tokenizer(BPE(unk_token="[UNK]")) 
trainer = BpeTrainer(special_tokens=["[UNK]", "[CLS]", "[SEP]", "[PAD]", "[MASK]"], vocab_size=1000) 
tokenizer.pre_tokenizer = Whitespace()

# 토크나이저 학습 
tokenizer.train_from_iterator(corpus, trainer)

# 테스트 
encoded = tokenizer.encode("자연어처리 모델을 학습해봅시다") 
print(f"토큰 ID: {encoded.ids}") 
print(f"토큰: {encoded.tokens}")

# 모델 저장 
tokenizer.save("bpe_korean.json")

# 학습된 어휘 확인 
vocab = tokenizer.get_vocab() 
print(f"어휘 크기: {len(vocab)}") 
print("상위 20개 토큰:") 
for token, id in sorted(vocab.items(), key=lambda x: x[1])[:20]:
```

```python
import numpy as np

from sklearn.manifold import TSNE

def plot_embeddings(model, word_list, filename=None):  
    # 단어 벡터 추출  
    word_vectors = np.array([model.wv[word] for word in word_list if word in model.wv])  
    words = [word for word in word_list if word in model.wv]

    # t-SNE로 차원 축소 
    tsne = TSNE(n_components=2, random_state=42) 
    embeddings_2d = tsne.fit_transform(word_vectors)

    # 시각화 
    plt.figure(figsize=(12, 8)) 
    plt.scatter(embeddings_2d[:, 0], embeddings_2d[:, 1], c='steelblue', alpha=0.5)

    for i, word in enumerate(words):  
        plt.annotate(word,  
                     xy=(embeddings_2d[i, 0], embeddings_2d[i, 1]),  
                     xytext=(5, 2),  
                     textcoords='offset points',  
                     fontsize=12)

    plt.title('Word Embeddings Visualization (t-SNE)') 
    plt.grid(True)

    if filename: 
        plt.savefig(filename) 
    plt.show()
```

## 페이지 47-48 - Hugging Face Transformers로 BERT 사용하기

```python
from transformers import BertModel, BertTokenizer 
import torch 
from scipy.spatial.distance import cosine

# 다국어 BERT 모델과 토크나이저 로드 
tokenizer = BertTokenizer.from_pretrained('bert-base-multilingual-cased') 
model = BertModel.from_pretrained('bert-base-multilingual-cased')

# 문장 리스트 
sentences = [ 
    "자연어 처리는 컴퓨터가 인간의 언어를 이해하는 분야입니다.", 
    "배를 타고 여행을 떠났습니다.", 
    "배가 너무 달아서 많이 먹었습니다." 
]

# 각 문장에 대한 BERT 임베딩 추출 
for sentence in sentences: 
    # 토큰화 및 모델 입력 준비 
    inputs = tokenizer(sentence, return_tensors="pt", padding=True, truncation=True)

    # BERT 모델로 임베딩 추출 
    with torch.no_grad(): 
        outputs = model(**inputs)

    # 문장 임베딩 (CLS 토큰의 마지막 은닉 상태) 
    sentence_embedding = outputs.last_hidden_state[:, 0, :].numpy()

    # 각 토큰의 임베딩 
    token_embeddings = outputs.last_hidden_state.numpy()

# 다의어 "배"의 문맥에 따른 임베딩 비교 
def get_word_embedding(sentence, word): 
    inputs = tokenizer(sentence, return_tensors="pt")
    with torch.no_grad(): 
        outputs = model(**inputs)
    tokens = tokenizer.convert_ids_to_tokens(inputs['input_ids'][0]) 
    try: 
        word_idx = tokens.index(word) 
        return outputs.last_hidden_state[0, word_idx, :].numpy() 
    except ValueError: 
        return None

# "배" 단어의 문맥별 임베딩 추출 
ship_embedding = get_word_embedding("배를 타고 여행을 떠났습니다.", "배") 
fruit_embedding = get_word_embedding("배가 너무 달아서 많이 먹었습니다.", "배")

if ship_embedding is not None and fruit_embedding is not None:  
    similarity = 1 - cosine(ship_embedding, fruit_embedding)
```

```python
# 파이썬 구현 예시 (간단한 버전) 
def noise_gate(signal, threshold=0.01, beta=0): 
    gated = np.copy(signal) 
    mask = np.abs(signal) < threshold 
    gated[mask] = beta * signal[mask] 
    return gated
```

## 페이지 83-85 - 음성 특징 시각화 (상세)

```python
import numpy as np 
import matplotlib.pyplot as plt 
import librosa 
import librosa.display

# --- 설정값(강의에서 설명 포인트) --- 
AUDIO_PATH = "your_audio.wav" # 실습용 음성 파일 경로 
SR_TARGET = 16_000 # 리샘플링 샘플레이트(음성 작업에서 자주 사용) 
N_FFT = 1024 # STFT 윈도우 크기 (≈ 64 ms @16kHz) 
HOP_LENGTH = 256 # 프레임 간격 (≈ 16 ms @16kHz) 
WIN_LENGTH = 1024 # 윈도우 길이 
N_MELS = 80 # 멜 필터 개수 (딥러닝 기본기) 
N_MFCC = 13 # MFCC 차원 (전통 특징 기본기) 
# ---

# (선택) 프리엠퍼시스: 고주파 강조 y_pre[n] = y[n] - α y[n-1] 
def pre_emphasis(y, alpha=0.97): 
    y_emph = np.append(y[0], y[1:] - alpha * y[:-1]) 
    return y_emph

# 로드 & (선택) 프리엠퍼시스 적용 
y, sr = librosa.load(AUDIO_PATH, sr=SR_TARGET) 
y_emph = pre_emphasis(y, alpha=0.97)

# --- 1) Waveform --- 
plt.figure(figsize=(10, 3)) 
librosa.display.waveshow(y_emph, sr=sr) 
plt.title("Waveform (pre-emphasized)") 
plt.xlabel("Time (s)") 
plt.ylabel("Amplitude") 
plt.tight_layout() 
plt.show()

# --- 2) Spectrogram (STFT) --- 
# STFT: X(m, w) = Σ_n x[n] w[n-m] e^[-j w n] 
S_complex = librosa.stft(y_emph, n_fft=N_FFT, hop_length=HOP_LENGTH, win_length=WIN_LENGTH, window="hann") 
S_mag = np.abs(S_complex) # |X| 
S_db = librosa.amplitude_to_db(S_mag, ref=np.max)

plt.figure(figsize=(10, 4)) 
librosa.display.specshow(S_db, sr=sr, hop_length=HOP_LENGTH, x_axis="time", y_axis="linear") 
plt.title("Spectrogram (|STFT| in dB)") 
plt.colorbar(format="%.1f dB") 
plt.tight_layout() 
plt.show()

# --- 3) Mel-Spectrogram --- 
# Mel 변환: m = 2595 * log10(1 + f/700) 
M = librosa.feature.melspectrogram(y=y_emph, sr=sr, n_fft=N_FFT, hop_length=HOP_LENGTH, n_mels=N_MELS, power=2.0) 
# power=2 -> |X|^2 
M_db = librosa.power_to_db(M, ref=np.max) 
plt.figure(figsize=(10, 4)) 
librosa.display.specshow(M_db, sr=sr, hop_length=HOP_LENGTH, x_axis="time", y_axis="mel") 
plt.title("Mel-Spectrogram (dB)") 
plt.colorbar(format="%.1f dB") 
plt.tight_layout() 
plt.show()

# --- 4) MFCC --- 
# c_k = Σ_{m=1..M} log(S_m) * cos[π k / M * (m - 0.5)] 
MFCC = librosa.feature.mfcc(S=librosa.power_to_db(M), n_mfcc=N_MFCC) 
plt.figure(figsize=(10, 4)) 
librosa.display.specshow(MFCC, x_axis="time", sr=sr, hop_length=HOP_LENGTH) 
plt.title(f"MFCC (n_mfcc={N_MFCC})") 
plt.ylabel("MFCC Coefficients") 
plt.colorbar() 
plt.tight_layout() 
plt.show()
```

## 페이지 97-100 - 어텐션 가중치 시각화

```python
import numpy as np 
import torch 
import torch.nn.functional as F 
import matplotlib.pyplot as plt

# (A) 어텐션 히트맵: 크로스 어텐션 [Layers, Heads, T_out, T_in] 
def plot_cross_attention_heatmap(attn, out_tokens=None, frame_times=None, layer=0, head=0, title="Cross-Attention Weights"):
    # attn: np.ndarray [L, H, T_out, T_in] 또는 torch.Tensor
    # out_tokens: 출력 토큰(문자/음소) 리스트 길이 = T_out (y축 라벨용)
    # frame_times: 각 입력 프레임의 초 단위 시간축(길이 T_in)

    if isinstance(attn, torch.Tensor):    
        attn = attn.detach().cpu().numpy()    
    A = attn[layer, head] # [T_out, T_in]

    plt.figure(figsize=(10, 6)) 
    plt.imshow(A, aspect="auto", origin="lower") 
    plt.colorbar() 
    plt.title(f"{title} (Layer {layer}, Head {head})") 
    plt.ylabel("Output tokens (t)") 
    plt.xlabel("Input frames (time)")

    # y축에 토큰 라벨 
    if out_tokens is not None: 
        yticks = np.arange(len(out_tokens)) 
        plt.yticks(yticks, out_tokens, fontsize=9)

    # x축을 시간으로 표시 (선택) 
    if frame_times is not None: 
        xticks = np.linspace(0, len(frame_times)-1, num=10, dtype=int) 
        xticklabs = [f"{frame_times[i]:.2f}s" for i in xticks] 
        plt.xticks(xticks, xticklabs, rotation=45)

    plt.tight_layout() 
    plt.show()

# --- (B) 멜 스펙트로그램 위 정렬 경로 오버레이 --- 
def plot_alignment_on_mel(mel_db, attn_one_layer_head, frame_times=None, out_tokens=None, title="Alignment on Mel"):  
    # mel_db: [F, T_in] (로그멜 dB 등) 2D  
    # attn_one_layer_head: [T_out, T_in] (단일 layer/head의 크로스 어텐션)  
    # frame_times: 길이 T_in, 초 단위  

    if isinstance(attn_one_layer_head, torch.Tensor):  
        attn_one_layer_head = attn_one_layer_head.detach().cpu().numpy()  
    if isinstance(mel_db, torch.Tensor):  
        mel_db = mel_db.detach().cpu().numpy()  

    # 정렬 경로 (각 출력 스텝의 argmax 프레임)  
    path = attn_one_layer_head.argmax(axis=1) # [T_out]  

    plt.figure(figsize=(10, 6))  
    plt.imshow(mel_db, aspect="auto", origin="lower")  
    plt.colorbar(label="dB")  
    plt.title(title)  
    plt.ylabel("Mel bins")  
    plt.xlabel("Time (frames)")

    # 시간축 라벨 (선택)  
    if frame_times is not None:    
        xticks = np.linspace(0, len(frame_times)-1, num=10, dtype=int)    
        xticklabs = [f"{frame_times[i]:.2f}s" for i in xticks]    
        plt.xticks(xticks, xticklabs, rotation=45)    

    # 정렬 경로 오버레이  
    t_out = np.arange(len(path))    
    ax = plt.gca().twinx()    
    ax.plot(path, t_out, linewidth=1)    
    ax.set_xlabel("Argmax frame per output step")    
    plt.tight_layout()    
    plt.show()

# --- (C) 토큰 신뢰도: max prob & entropy 곡선 --- 
def plot_token_confidence(logits, out_tokens=None, title="Token Confidence / Entropy"):  
    if isinstance(logits, torch.Tensor):  
        logits = logits.detach().cpu()  
    if logits.ndim == 3:  
        logits = logits[0] # B=1 가정  

    probs = F.softmax(torch.as_tensor(logits), dim=-1).numpy() # [T_out, V]  
    max_p = probs.max(axis=1) # [T_out]  

    # 엔트로피 H = -sum p log p  
    eps = 1e-12  
    entropy = -(probs * np.log(probs + eps)).sum(axis=1)  

    t = np.arange(len(max_p))  
    fig = plt.figure(figsize=(10, 5))  
    ax1 = plt.gca()  
    ax1.plot(t, max_p, label="Max Prob")  
    ax1.set_xlabel("Output step")  
    ax1.set_ylabel("Max Probability")  
    ax1.set_title(title)  

    ax2 = ax1.twinx()  
    ax2.plot(t, entropy, label="Entropy", linestyle="--")  
    ax2.set_ylabel("Entropy")  

    # 토큰 라벨 (선택)  
    if out_tokens is not None:  
        xticks = np.arange(len(out_tokens))  
        ax1.set_xticks(xticks)  
        ax1.set_xticklabels(out_tokens, rotation=45, fontsize=9)  

    fig.tight_layout()  
    plt.show()
```
